#function score query 人工干预排名
GET index01/_search
{
  "query": {
    "function_score": {
      "query": {
        "range": {
          "age": {
            "gte": 0,
            "lte": 99
          }
        }
      },
      "functions": [
        {
          "filter": {
            "term": {
              "name.firstname": "文"
            }
          },
          "weight": 2
        }
      ],
      "boost_mode": "sum"
    }
  }
}

#bool复合查询 must和should参与算分 filter和must_not不参与算分 性能更好
GET index01/_search
{
  "query": {
    "bool": {
      "must": [
        {
          "term": {
            "name.firstname": {
              "value": "文"
            }
          }
        }
      ],
      "must_not": [
        {
          "exists": {
            "field": "telphone"
          }
        }
      ],
      "filter": [
        {
          "range": {
            "age": {
              "gte": 10,
              "lte": 99
            }
          }
        }
      ],
      "should": [
        {
          "term": {
            "info": {
              "value": "三国"
            }
          }
        },
        {
          "term": {
            "info": {
              "value": "战国"
            }
          }
        }
      ]
    }
  }
}
